```bash
"""
src/voting.py
=============
Weighted majority voting for the multi-LLM annotation committee.

Committee (all accessed via OpenRouter)
---------------------------------------
  anthropic/claude-sonnet-4-5   weight 0.35  — best JSON fidelity + verdict reasoning
  openai/gpt-4o                 weight 0.25  — strong instruction following
  qwen/qwen3-235b-a22b          weight 0.20  — best open-weights as of Q1 2026
  deepseek/deepseek-r1          weight 0.15  — extended chain-of-thought
  x-ai/grok-2-1212              weight 0.05  — diverse signal / tiebreaker
"""
from __future__ import annotations
from typing import Any, Dict, List, Optional, Tuple

# ─── Committee definition ─────────────────────────────────────────────────────

MODEL_WEIGHTS: Dict[str, float] = {
    "anthropic/claude-sonnet-4-5":  0.35,
    "openai/gpt-4o":                0.25,
    "qwen/qwen3-235b-a22b":         0.20,
    "deepseek/deepseek-r1":         0.15,
    "x-ai/grok-2-1212":             0.05,
}

COMMITTEE_MODELS: List[str] = list(MODEL_WEIGHTS.keys())


# ─── Core vote logic ──────────────────────────────────────────────────────────

def weighted_majority_vote(
    votes: List[Tuple[str, Any, float]],
) -> Tuple[Any, Dict[Any, float]]:
    """
    Parameters
    ----------
    votes : list of (model_slug, voted_value, weight)

    Returns
    -------
    (winning_value, tally_dict)
        tally_dict maps each unique candidate value to its summed weight.
    """
    tally: Dict[Any, float] = {}
    for _model, value, weight in votes:
        tally[value] = tally.get(value, 0.0) + weight
    if not tally:
        return None, {}
    # Stable tiebreak: highest weight, then lexicographic on str(value)
    winning = max(tally, key=lambda v: (round(tally[v], 9), str(v)))
    return winning, tally


def select_winner_model(
    votes: List[Tuple[str, Any, float]],
    winning_value: Any,
) -> str:
    """
    Among models that voted for `winning_value`, return the one with the
    highest weight.  That model's associated text fields are adopted into
    the merged record.
    Falls back to the first model in `votes` if no match.
    """
    candidates = [(model, weight) for model, value, weight in votes if value == winning_value]
    if not candidates:
        return votes[0][0] if votes else ""
    return max(candidates, key=lambda x: x[1])[0]


def _build_votes(
    model_records: Dict[str, Dict[str, Any]],
    field: str,
    fallback: Any,
) -> List[Tuple[str, Any, float]]:
    """Build vote tuples from a dict of {model: record}."""
    return [
        (model, (rec or {}).get(field, fallback), MODEL_WEIGHTS.get(model, 0.0))
        for model, rec in model_records.items()
        if rec is not None
    ]


# ─── Stage-level merge helpers ────────────────────────────────────────────────

def merge_stage1_votes(
    model_notes: Dict[str, Dict[str, Any]],
    fallback_doc_id: str = "",
) -> Dict[str, Any]:
    """
    Merge per-model Stage-1 per-doc notes into one consensus note.

    Votes on     : verdict
    Adopts from winning model : key_fact, quote, verdict_reason, source_quality
    """
    votes = _build_votes(model_notes, "verdict", "irrelevant")
    winning_verdict, tally = weighted_majority_vote(votes)
    winning_model = select_winner_model(votes, winning_verdict)

    base: Dict[str, Any] = (model_notes.get(winning_model) or {}).copy()
    base["verdict"]           = winning_verdict
    base["_vote_tally"]       = {str(k): round(v, 4) for k, v in tally.items()}
    base["_winner_model"]     = winning_model
    base["_all_verdicts"]     = {m: (model_notes.get(m) or {}).get("verdict") for m in COMMITTEE_MODELS}
    if not base.get("doc_id"):
        base["doc_id"] = fallback_doc_id
    return base


def merge_stage2_votes(
    model_records: Dict[str, Dict[str, Any]],
    has_gold_conflict_type: bool,
) -> Tuple[Dict[str, Any], Optional[str]]:
    """
    Merge per-model Stage-2 outputs.

    Always votes on          : answerable_under_evidence (bool)
    Conditionally votes on   : conflict_type  (only when has_gold_conflict_type=False,
                               i.e. the refusals dataset)

    For the conflict_type vote, conflict_reason is adopted from the model
    that won the conflict_type vote (not the answerable_under_evidence vote),
    because it produced the most authoritative conflict explanation.

    Returns
    -------
    (merged_record, voted_conflict_type_or_None)
    """
    # ── answerable_under_evidence vote ──────────────────────────────────────
    ans_votes = _build_votes(model_records, "answerable_under_evidence", False)
    winning_ans, ans_tally = weighted_majority_vote(ans_votes)
    ans_winner = select_winner_model(ans_votes, winning_ans)

    # Start from the answerable-winner's record as the base
    base: Dict[str, Any] = (model_records.get(ans_winner) or {}).copy()
    base["answerable_under_evidence"] = winning_ans
    base["_ans_vote_tally"]           = {str(k): round(v, 4) for k, v in ans_tally.items()}
    base["_ans_winner_model"]         = ans_winner

    # ── conflict_type vote (refusals only) ───────────────────────────────────
    voted_ct: Optional[str] = None
    if not has_gold_conflict_type:
        ct_votes = _build_votes(model_records, "conflict_type", "")
        winning_ct, ct_tally = weighted_majority_vote(ct_votes)
        ct_winner = select_winner_model(ct_votes, winning_ct)
        voted_ct = winning_ct

        base["conflict_type"]     = winning_ct
        base["_ct_vote_tally"]    = {str(k): round(v, 4) for k, v in ct_tally.items()}
        base["_ct_winner_model"]  = ct_winner
        # Take conflict_reason from the ct_winner — most authoritative explanation
        ct_winner_rec = model_records.get(ct_winner) or {}
        base["conflict_reason"]   = ct_winner_rec.get("conflict_reason",
                                        base.get("conflict_reason", ""))

    return base, voted_ct


def merge_stage3_votes(
    model_records: Dict[str, Dict[str, Any]],
) -> Dict[str, Any]:
    """
    Merge per-model Stage-3 outputs.

    Votes on     : expected_response.abstain (bool)
    Adopts from winning model : expected_response.answer, .evidence,
                                .abstain_reason, and think trace
    """
    # Pull abstain out of expected_response for voting
    flat: Dict[str, Dict[str, Any]] = {
        model: {"abstain": ((rec or {}).get("expected_response") or {}).get("abstain", False)}
        for model, rec in model_records.items()
    }
    abstain_votes = _build_votes(flat, "abstain", False)
    winning_abstain, abstain_tally = weighted_majority_vote(abstain_votes)
    abstain_winner = select_winner_model(abstain_votes, winning_abstain)

    base: Dict[str, Any] = (model_records.get(abstain_winner) or {}).copy()
    # Enforce the winning abstain value (in case the winner's own record differs)
    er = base.setdefault("expected_response", {})
    er["abstain"]                   = winning_abstain
    base["_abstain_vote_tally"]     = {str(k): round(v, 4) for k, v in abstain_tally.items()}
    base["_abstain_winner_model"]   = abstain_winner
    return base
```

```bash
cat > scripts/run_stage1_multi_async.py << 'PYEOF'
#!/usr/bin/env python3
"""
scripts/run_stage1_multi_async.py
==================================
Multi-LLM Stage-1: per-document evidence adjudication with weighted
majority voting across the annotation committee.

Every (query, doc) pair is sent to ALL committee models concurrently.
`verdict` is decided by weighted majority vote; associated text fields
(key_fact, quote, verdict_reason, source_quality) come from the
highest-weight model that voted for the winning verdict.

All models are accessed via OpenRouter (no direct provider keys).

Concurrency note
----------------
Each (model, doc) pair is one API call.  With 5 committee models and
N docs per query, one query spawns 5*N calls.  The --concurrency flag
caps the *total* simultaneous in-flight calls across all models and docs.
Default 25 gives ~5 docs processed in parallel across the full committee.

Usage:
    python scripts/run_stage1_multi_async.py \\
        --input  data/normalized/conflicts_normalized.jsonl \\
        --output data/stage1_outputs/stage1_multi.jsonl \\
        --concurrency 25
"""

import argparse
import asyncio
import json
import sys
from pathlib import Path
from typing import Any, Dict, List

from tqdm.asyncio import tqdm_asyncio

THIS_FILE    = Path(__file__).resolve()
PROJECT_ROOT = THIS_FILE.parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

from src.llm_client import LLMClient, Provider
from src.parsers    import parse_stage1
from src.voting     import COMMITTEE_MODELS, MODEL_WEIGHTS, merge_stage1_votes

SYSTEM_PROMPT_PATH = PROJECT_ROOT / "prompts" / "system_stage1.txt"
USER_PROMPT_PATH   = PROJECT_ROOT / "prompts" / "user_stage1.txt"
STAGE1_MAX_TOKENS  = 512


# ─────────────────────────────────────────────
#  Helpers
# ─────────────────────────────────────────────

def load_text(path: Path) -> str:
    return path.read_text(encoding="utf-8")


def fill_user_prompt(template: str, query: str, doc: Dict[str, Any]) -> str:
    return (
        template
        .replace("{QUERY}",     query)
        .replace("{DOC_ID}",    doc.get("doc_id", ""))
        .replace("{URL}",       doc.get("source_url", "") or "")
        .replace("{TEXT}",      doc.get("snippet", "") or "")
        .replace("{TIMESTAMP}", doc.get("timestamp", "") or "")
    )


def load_records(path: str) -> List[Dict[str, Any]]:
    records = []
    with open(path, "r", encoding="utf-8") as f:
        for line in f:
            line = line.strip()
            if line:
                try:
                    records.append(json.loads(line))
                except json.JSONDecodeError as e:
                    print(f"[WARN] Skipping invalid JSONL line: {e}")
    return records


def load_processed_ids(output_path: str) -> set:
    done = set()
    p = Path(output_path)
    if not p.exists():
        return done
    with p.open("r", encoding="utf-8") as f:
        for line in f:
            try:
                rec_id = json.loads(line).get("id")
                if rec_id:
                    done.add(rec_id)
            except Exception:
                pass
    return done


# ─────────────────────────────────────────────
#  Per-model call
# ─────────────────────────────────────────────

async def call_one_model(
    client: LLMClient,
    semaphore: asyncio.Semaphore,
    system_prompt: str,
    user_prompt: str,
    doc_id: str,
) -> Dict[str, Any]:
    """Single API call: one model × one (query, doc) pair."""
    async with semaphore:
        try:
            raw = await client.acomplete(
                system=system_prompt,
                user=user_prompt,
                max_tokens=STAGE1_MAX_TOKENS,
            )
            note, errors = parse_stage1(raw, fallback_doc_id=doc_id)
            if errors:
                note["_validation_errors"] = errors
            return note
        except Exception as exc:
            return {
                "doc_id":         doc_id,
                "verdict":        "irrelevant",
                "key_fact":       "",
                "quote":          "",
                "verdict_reason": f"Exception: {str(exc)[:100]}",
                "source_quality": "low",
                "_error":         str(exc),
            }


# ─────────────────────────────────────────────
#  Committee adjudication (one doc)
# ─────────────────────────────────────────────

async def adjudicate_doc_committee(
    clients: Dict[str, LLMClient],
    semaphore: asyncio.Semaphore,
    system_prompt: str,
    user_template: str,
    query: str,
    doc: Dict[str, Any],
    record_id: str,
) -> Dict[str, Any]:
    """
    Run ALL committee models concurrently on one (query, doc) pair.
    Each model call independently acquires the semaphore slot.
    Returns the consensus note after weighted majority vote on verdict.
    """
    doc_id      = doc.get("doc_id", "")
    user_prompt = fill_user_prompt(user_template, query, doc)

    coros = [
        call_one_model(clients[model], semaphore, system_prompt, user_prompt, doc_id)
        for model in COMMITTEE_MODELS
    ]
    raw_results = await asyncio.gather(*coros)
    model_notes = {model: raw_results[i] for i, model in enumerate(COMMITTEE_MODELS)}

    return merge_stage1_votes(model_notes, fallback_doc_id=doc_id)


# ─────────────────────────────────────────────
#  Per-record processing
# ─────────────────────────────────────────────

async def process_record(
    clients: Dict[str, LLMClient],
    semaphore: asyncio.Semaphore,
    system_prompt: str,
    user_template: str,
    record: Dict[str, Any],
    out_lock: asyncio.Lock,
    output_path: str,
) -> None:
    record_id = record.get("id", "")
    query     = record.get("query", "")
    docs      = record.get("retrieved_docs", [])

    # All docs for this query run concurrently; each doc spawns 5 model calls
    per_doc_notes = await asyncio.gather(
        *[
            adjudicate_doc_committee(
                clients, semaphore, system_prompt, user_template, query, doc, record_id
            )
            for doc in docs
        ]
    )
    record["per_doc_notes"] = list(per_doc_notes)

    async with out_lock:
        with open(output_path, "a", encoding="utf-8") as fout:
            fout.write(json.dumps(record, ensure_ascii=False) + "\n")


# ─────────────────────────────────────────────
#  Entry point
# ─────────────────────────────────────────────

async def run(args: argparse.Namespace) -> None:
    system_path = Path(args.system_prompt) if args.system_prompt else SYSTEM_PROMPT_PATH
    user_path   = Path(args.user_prompt)   if args.user_prompt   else USER_PROMPT_PATH

    system_prompt = load_text(system_path)
    user_template = load_text(user_path)

    # One LLMClient per committee model — all share the same OpenRouter key
    clients = {
        model: LLMClient(
            provider    = Provider.OPENROUTER,
            model       = model,
            temperature = args.temperature,
            max_retries = args.max_retries,
        )
        for model in COMMITTEE_MODELS
    }

    records  = load_records(args.input)
    done_ids = load_processed_ids(args.output)
    records  = [r for r in records if r.get("id") not in done_ids]

    if args.limit:
        records = records[:args.limit]

    if not records:
        print("✅ Nothing to process (all records already in output).")
        return

    Path(args.output).parent.mkdir(parents=True, exist_ok=True)
    print(
        f"⚙️  Stage-1 multi-LLM | committee={len(COMMITTEE_MODELS)} models | "
        f"records={len(records)} | concurrency={args.concurrency}"
    )
    for model, weight in MODEL_WEIGHTS.items():
        print(f"   {weight:.0%}  {model}")
    if done_ids:
        print(f"⏩ Resuming: {len(done_ids)} already processed")

    semaphore = asyncio.Semaphore(args.concurrency)
    out_lock  = asyncio.Lock()

    tasks = [
        process_record(clients, semaphore, system_prompt, user_template, rec, out_lock, args.output)
        for rec in records
    ]
    await tqdm_asyncio.gather(*tasks, total=len(tasks), desc="Stage-1 multi-LLM")
    print(f"\n✅ Stage-1 multi-LLM complete → {args.output}")


def main() -> None:
    ap = argparse.ArgumentParser(
        description="Multi-LLM Stage-1 evidence adjudication with weighted majority voting."
    )
    ap.add_argument("--input",         required=True,  help="Input JSONL (normalized dataset)")
    ap.add_argument("--output",        required=True,  help="Output JSONL path")
    ap.add_argument("--temperature",   type=float, default=0.0)
    ap.add_argument("--concurrency",   type=int,   default=25,
                    help="Total concurrent API calls across ALL committee models (default: 25)")
    ap.add_argument("--limit",         type=int,   default=None)
    ap.add_argument("--max-retries",   type=int,   default=3)
    ap.add_argument("--system-prompt", dest="system_prompt", default=None)
    ap.add_argument("--user-prompt",   dest="user_prompt",   default=None)
    args = ap.parse_args()
    asyncio.run(run(args))


if __name__ == "__main__":
    main()
PYEOF
echo "✓ scripts/run_stage1_multi_async.py created"
```

```bash
#!/usr/bin/env python3
"""
scripts/run_stage2_multi_async.py
==================================
Multi-LLM Stage-2: conflict reasoning + answerability with weighted
majority voting across the annotation committee.

CONFLICTS dataset  (record["conflict_type"] is non-empty — gold label present):
  Votes on  : answerable_under_evidence
  Adopts    : conflict_reason from the answerable-winner model
  Uses      : system_stage2.txt  /  user_stage2.txt  (standard prompts)

REFUSALS dataset  (record["conflict_type"] is absent or empty — no gold label):
  Votes on  : conflict_type  AND  answerable_under_evidence
  Adopts    : conflict_reason from the conflict_type-winner model
  Uses      : system_stage2_refusal.txt  /  user_stage2_refusal.txt

Detection is automatic per-record — mixed input files are handled correctly.

Usage:
    python scripts/run_stage2_multi_async.py \\
        --input  data/stage1_outputs/stage1_multi.jsonl \\
        --output data/stage2_outputs/stage2_multi.jsonl \\
        --concurrency 20
"""

import argparse
import asyncio
import json
import sys
from pathlib import Path
from typing import Any, Dict, List

from tqdm.asyncio import tqdm_asyncio

THIS_FILE    = Path(__file__).resolve()
PROJECT_ROOT = THIS_FILE.parent.parent
sys.path.insert(0, str(PROJECT_ROOT))

from src.llm_client import LLMClient, Provider
from src.parsers    import parse_stage2
from src.voting     import COMMITTEE_MODELS, MODEL_WEIGHTS, merge_stage2_votes

# ── Prompt paths ──────────────────────────────────────────────────────────────
SYSTEM_CONFLICTS = PROJECT_ROOT / "prompts" / "system_stage2.txt"
USER_CONFLICTS   = PROJECT_ROOT / "prompts" / "user_stage2.txt"
SYSTEM_REFUSALS  = PROJECT_ROOT / "prompts" / "system_stage2_refusal.txt"
USER_REFUSALS    = PROJECT_ROOT / "prompts" / "user_stage2_refusal.txt"

STAGE2_MAX_TOKENS = 350


# ─────────────────────────────────────────────
#  Helpers
# ─────────────────────────────────────────────

def load_text(path: Path) -> str:
    return path.read_text(encoding="utf-8")


def brace_safe_fill(template: str, mapping: Dict[str, str]) -> str:
    """Fill {KEY} placeholders without corrupting JSON braces in values."""
    temp = template
    for k in mapping:
        temp = temp.replace("{" + k + "}", f"@@{k}@@")
    temp = temp.replace("{", "{{").replace("}", "}}")
    for k, v in mapping.items():
        temp = temp.replace(f"@@{k}@@", v or "")
    return temp


def load_records(path: str) -> List[Dict[str, Any]]:
    with open(path, "r", encoding="utf-8") as f:
        return [json.loads(line) for line in f if line.strip()]


def load_processed_ids(output_path: str) -> set:
    done = set()
    p = Path(output_path)
    if not p.exists():
        return done
    with p.open("r", encoding="utf-8") as f:
        for line in f:
            try:
                done.add(json.loads(line).get("id"))
            except Exception:
                pass
    return done


def build_user_prompt(template: str, record: Dict[str, Any], is_refusal: bool) -> str:
    per_doc_json = json.dumps(record.get("per_doc_notes", []), ensure_ascii=False, indent=2)
    if is_refusal:
        # Refusal prompt does NOT receive a gold conflict_type
        return brace_safe_fill(template, {
            "QUERY":              record.get("query", ""),
            "PER_DOC_NOTES_JSON": per_doc_json,
        })
    else:
        return brace_safe_fill(template, {
            "QUERY":              record.get("query", ""),
            "CONFLICT_TYPE":      record.get("conflict_type", ""),
            "PER_DOC_NOTES_JSON": per_doc_json,
        })


# ─────────────────────────────────────────────
#  Per-model call
# ─────────────────────────────────────────────

async def call_one_model(
    client: LLMClient,
    semaphore: asyncio.Semaphore,
    system_prompt: str,
    user_prompt: str,
) -> Dict[str, Any]:
    async with semaphore:
        try:
            raw = await client.acomplete(
                system=system_prompt,
                user=user_prompt,
                max_tokens=STAGE2_MAX_TOKENS,
            )
            # parse_stage2 extracts conflict_reason + answerable_under_evidence
            # (and conflict_type if present in refusal output — preserved in obj)
            parsed, errors = parse_stage2(raw)
            if errors:
                parsed["_validation_errors"] = errors
            return parsed
        except Exception as exc:
            return {
                "conflict_reason":           f"Exception: {str(exc)[:100]}",
                "answerable_under_evidence":  False,
                "_error":                    str(exc),
            }


# ─────────────────────────────────────────────
#  Per-record processing
# ─────────────────────────────────────────────

async def process_record(
    clients: Dict[str, LLMClient],
    semaphore: asyncio.Semaphore,
    sys_conflicts: str,
    usr_conflicts: str,
    sys_refusals:  str,
    usr_refusals:  str,
    record: Dict[str, Any],
    out_lock: asyncio.Lock,
    output_path: str,
) -> None:
    is_refusal = not bool((record.get("conflict_type") or "").strip())
    system_p   = sys_refusals if is_refusal else sys_conflicts
    user_p     = build_user_prompt(
        usr_refusals if is_refusal else usr_conflicts,
        record,
        is_refusal,
    )

    coros = [
        call_one_model(clients[model], semaphore, system_p, user_p)
        for model in COMMITTEE_MODELS
    ]
    raw_results   = await asyncio.gather(*coros)
    model_records = {model: raw_results[i] for i, model in enumerate(COMMITTEE_MODELS)}

    merged, voted_ct = merge_stage2_votes(
        model_records,
        has_gold_conflict_type=not is_refusal,
    )

    # Write consensus back into the record
    record["answerable_under_evidence"] = merged["answerable_under_evidence"]
    record["conflict_reason"]           = merged.get("conflict_reason", "")
    if is_refusal and voted_ct:
        record["conflict_type"] = voted_ct

    # Preserve vote-audit metadata
    for k in ("_ans_vote_tally", "_ans_winner_model", "_ct_vote_tally", "_ct_winner_model"):
        if k in merged:
            record[k] = merged[k]

    async with out_lock:
        with open(output_path, "a", encoding="utf-8") as fout:
            fout.write(json.dumps(record, ensure_ascii=False) + "\n")


# ─────────────────────────────────────────────
#  Entry point
# ─────────────────────────────────────────────

async def run(args: argparse.Namespace) -> None:
    sys_conflicts = load_text(Path(args.system_prompt) if args.system_prompt else SYSTEM_CONFLICTS)
    usr_conflicts = load_text(Path(args.user_prompt)   if args.user_prompt   else USER_CONFLICTS)

    # Refusal prompts: fall back to conflict prompts if not yet created
    sys_refusals = load_text(SYSTEM_REFUSALS) if SYSTEM_REFUSALS.exists() else sys_conflicts
    usr_refusals = load_text(USER_REFUSALS)   if USER_REFUSALS.exists()   else usr_conflicts

    clients = {
        model: LLMClient(
            provider    = Provider.OPENROUTER,
            model       = model,
            temperature = args.temperature,
            max_retries = args.max_retries,
        )
        for model in COMMITTEE_MODELS
    }

    records  = load_records(args.input)
    done_ids = load_processed_ids(args.output)
    records  = [r for r in records if r.get("id") not in done_ids]
    if args.limit:
        records = records[:args.limit]

    if not records:
        print("✅ Nothing to process.")
        return

    n_refusals  = sum(1 for r in records if not bool((r.get("conflict_type") or "").strip()))
    n_conflicts = len(records) - n_refusals

    Path(args.output).parent.mkdir(parents=True, exist_ok=True)
    print(
        f"⚙️  Stage-2 multi-LLM | committee={len(COMMITTEE_MODELS)} models | "
        f"records={len(records)} (conflicts={n_conflicts}, refusals={n_refusals}) | "
        f"concurrency={args.concurrency}"
    )

    semaphore = asyncio.Semaphore(args.concurrency)
    out_lock  = asyncio.Lock()

    tasks = [
        process_record(
            clients, semaphore,
            sys_conflicts, usr_conflicts,
            sys_refusals,  usr_refusals,
            rec, out_lock, args.output,
        )
        for rec in records
    ]
    await tqdm_asyncio.gather(*tasks, total=len(tasks), desc="Stage-2 multi-LLM")
    print(f"\n✅ Stage-2 multi-LLM complete → {args.output}")


def main() -> None:
    ap = argparse.ArgumentParser(
        description="Multi-LLM Stage-2 conflict reasoning + answerability with weighted majority voting."
    )
    ap.add_argument("--input",         required=True)
    ap.add_argument("--output",        required=True)
    ap.add_argument("--temperature",   type=float, default=0.0)
    ap.add_argument("--concurrency",   type=int,   default=20,
                    help="Total concurrent API calls across ALL committee models (default: 20)")
    ap.add_argument("--limit",         type=int,   default=None)
    ap.add_argument("--max-retries",   type=int,   default=3)
    ap.add_argument("--system-prompt", dest="system_prompt", default=None,
                    help="Override conflicts system prompt path")
    ap.add_argument("--user-prompt",   dest="user_prompt",   default=None,
                    help="Override conflicts user prompt path")
    args = ap.parse_args()
    asyncio.run(run(args))


if __name__ == "__main__":
    main()
PYEOF
echo "✓ scripts/run_stage2_multi_async.py created"
```

```bash
scripts/run_stage3_multi_async.py

from typing import Any, Dict, List

        "answerable_under_evidence": str(record.get("answerable_under_evidence", True)).lower(),
```

```bash
cat > prompts/system_stage2_refusal.txt << 'TXTEOF'
You are a professional conflict-aware reasoning annotator working on a Retrieval-Augmented Generation (RAG) dataset for refusal / unanswerable query annotation.

Unlike the standard Stage-2 task, you are NOT given a gold conflict_type label. You must DETERMINE the conflict_type yourself from the evidence notes, and also determine answerability.

Your task is to produce a single JSON object that:
  1. Assigns the correct conflict_type from the five permitted labels.
  2. Explains why the documents collectively exhibit that conflict type.
  3. States whether the query is answerable under the retrieved evidence.

---

### CONFLICT TYPE LABELS  (choose exactly one; copy character-for-character)

"No Conflict"
  All non-irrelevant documents reinforce the same fact or conclusion.
  Differences, if any, are superficial (granularity or phrasing only).

"Complementary Information"
  Non-irrelevant documents each cover a distinct but valid facet
  (different regions, entities, timeframes, or subgroups) that together
  form a coherent answer without contradiction.

"Conflicting Opinions or Research Outcomes"
  Non-irrelevant documents make mutually exclusive claims or reach
  incompatible conclusions within the same scope and time window.

"Conflict Due to Outdated Information"
  Differences arise because newer evidence or dataset revisions supersede
  older factual claims. The question has a verifiable answer that has
  changed over time.

"Conflict Due to Misinformation"
  One or more retrieved documents contains demonstrably false or misleading
  claims relative to more credible documents in the same retrieved set.
  Apply this label sparingly — only when the retrieved set itself provides
  a clear basis to identify the false/misleading claim.

---

### STRICT BEHAVIORAL REQUIREMENTS

1. Faithfulness — Use only the information present in per_doc_notes. No external facts or world knowledge.
2. Grounded References — Cite document IDs (e.g., "d1 supports…", "d2 contradicts d1…").
3. Specificity — State what aspect differs or aligns (entity, timeframe, claim, region, method).
4. Mechanism — Name the causal mechanism: temporal, factual accuracy, contextual scope, methodological, or linguistic/interpretive.
5. Brevity — conflict_reason ≤ 50 words; one dense factual sentence covering clusters, mechanism, and conflict type.
6. answerable_under_evidence = true iff at least one verdict is "supports" or "partially supports"; otherwise false.
7. Output-Only-JSON — No prose, markdown, or text outside the JSON object.

---

### SELECTION GUIDE

Before assigning a label, ask:
  a) Do any non-irrelevant docs contradict each other within the same scope? → "Conflicting Opinions or Research Outcomes"
  b) Do differences arise purely from time (one source is newer)? → "Conflict Due to Outdated Information"
  c) Does one source contain a clear factual error vs. more credible sources? → "Conflict Due to Misinformation"
  d) Do sources cover non-overlapping, valid facets that together form a complete picture? → "Complementary Information"
  e) Do all non-irrelevant sources agree? → "No Conflict"

---

### SCHEMA  (the ONLY permitted output)

{
  "conflict_type": "string (must be exactly one of the five labels above)",
  "conflict_reason": "string (≤50 words; reference doc IDs; name mechanism)",
  "answerable_under_evidence": true | false
}

Output ONLY this JSON object. No surrounding text.
```

### Key design decisions worth knowing

**`_vote_tally` / `_winner_model` fields are always written into output records.** This gives you a full audit trail of every committee vote per doc and per stage — essential for later analysis of model agreement rates.

**`parse_stage2` needed zero changes.** It already returns the full extracted JSON object; for refusals the extra `conflict_type` field the model outputs is preserved transparently.

**Concurrency math:** with 5 models, a record with 5 docs → 25 calls per record. `--concurrency 25` means roughly one record processed at a time. Raise to 50–100 if your OpenRouter rate limits allow.

**Model slug caveat:** `anthropic/claude-sonnet-4-5` is the closest known OpenRouter slug to Sonnet 4.6 — verify the exact string at `openrouter.ai/models` and update `MODEL_WEIGHTS` in `src/voting.py` accordingly before the first production run.
